    def rotate(self):
        """rotar """
        if self.facing == RIGHT:
            self.facing = DOWN
        if self.facing == DOWN:
            self.facing = LEFT
        if self.facing == LEFT:
            self.facing = UP
        if self.facing == UP:
            self.facing = RIGHT
